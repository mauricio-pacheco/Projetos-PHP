function printCheck(){
	var msg;
	if ((navigator.appName.indexOf("Microsoft") != -1) && (navigator.appVersion.indexOf("Mac") != -1)) { /* Check for Internet Explorer and Macintosh */
		msg = "O Internet Explorer do Macintosh não oferece esta opção.\nPara imprimir esta página, segure a tecla Command e pressione P, ou\nselecione Imprimir... no menu arquivo.";
		alert(msg);
		}
	else {
		window.print();
		}
	return;
	}
